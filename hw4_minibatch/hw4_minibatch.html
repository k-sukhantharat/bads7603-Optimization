def search_direction(x, k, i):
    b = B[i:i+k]
    c = C[i:i+k]
    return - np.sum(b + (c * x))

def mini_batch(N,k,t):
    epoch = 5
    # initial x_previous
    x_prev = 1   
    f_hist = [f(x_prev)]
    for e in range(epoch):
        i = 0
        for n in range(1,N//k + 1):
            x = x_prev + (t * (search_direction(x_prev, k, i)))   
            f_hist.append(f(x))    
            x_prev = x
            i += k
    return f_hist

def result(N,t):

    k = 1 #batch_size
